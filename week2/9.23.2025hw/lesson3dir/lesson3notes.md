# navigating and creating directories

*   cd = change directory

*   mkdir = make directory

*   / = directory

*   ~ = root

*   mkdir -p paentdir/childdir {1..100} = makes 100 folders in childdir

*   pwd = present working directory

*   cd / = root directory

*   rm = remove file (-rf) = recursively (-f) = force

*   $HOME = root alias

*  rmdir = remove directory (*) = wildacrd to remove all directories ex: rmdir /directory/*
